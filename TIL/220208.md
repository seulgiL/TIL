```python
  a = input()
  b = input()
for i in range(len(b)):
      #str인 b의 길이를 통해 인덱스 접근 / 인덱스 마지막 값 부터 접근 len(b)==3 이면 필요한 인덱스는 2부터. 
      print(int(b[len(b)-i-1]) * int(a))
  ```
